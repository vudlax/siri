# Build many-to-many training windows (autoencoding)
def build_windows(seq, seq_len=64):
    """Return X, Y for autoencoding: input and output are the same sequence windows."""
    X, Y = [], []
    for i in range(0, len(seq) - seq_len + 1):
        chunk = seq[i:i+seq_len]
        X.append(chunk)
        Y.append(chunk)   # autoencoder target = input (reconstruct)
    return np.array(X, dtype=np.int32), np.array(Y, dtype=np.int32)

# ================== MNIST Undercomplete Autoencoder ==================
import numpy as np
import matplotlib.pyplot as plt
from keras.datasets import mnist
from tensorflow.keras import layers, models, callbacks

# 1) Load MNIST
(X_train, y_train), (X_test, y_test) = mnist.load_data()

# 2) Scale to [0,1] and flatten to vectors (28*28=784)
X_train = (X_train.astype("float32") / 255.0).reshape((-1, 28*28))
X_test  = (X_test.astype("float32")  / 255.0).reshape((-1, 28*28))

# 3) Build an undercomplete autoencoder
def build_autoencoder(latent_dim=8):
    inp = layers.Input(shape=(784,))
    # Encoder
    x = layers.Dense(256, activation="relu")(inp)
    x = layers.Dense(128, activation="relu")(x)
    z = layers.Dense(latent_dim, activation=None, name="latent")(x)  # codings

    # Decoder
    x = layers.Dense(128, activation="relu")(z)
    x = layers.Dense(256, activation="relu")(x)
    out = layers.Dense(784, activation="sigmoid")(x)  # reconstruct pixel probs [0,1]

    autoenc = models.Model(inp, out, name=f"ae_latent{latent_dim}")
    autoenc.compile(optimizer="adam", loss="mse")  # MSE works well for MNIST
    return autoenc

# Choose a small latent size (codings)
LATENT_DIM = 8   # try 8 first;

autoenc = build_autoencoder(LATENT_DIM)
autoenc.summary()

# 4) Train with validation split + early stopping
es = callbacks.EarlyStopping(monitor="val_loss", patience=3, restore_best_weights=True)
hist = autoenc.fit(
    X_train, X_train,           # autoencoder: input == target
    epochs=30, batch_size=256,
    validation_split=0.1,
    callbacks=[es],
    verbose=1
)

# ================= IMDb Undercomplete Seq Autoencoder (Teacher Forcing, Fast, no Keras mask) =================
import numpy as np, tensorflow as tf, matplotlib.pyplot as plt
from keras.datasets import imdb
from keras.preprocessing import sequence
from tensorflow.keras import layers, models, callbacks

# ---------------- Data ----------------
max_features = 10000
maxlen = 200
(X_train, _), (X_test, _) = imdb.load_data(num_words=max_features)
X_train = sequence.pad_sequences(X_train, maxlen=maxlen, padding='post')
X_test  = sequence.pad_sequences(X_test,  maxlen=maxlen, padding='post')

# (Optional) speed: train on a subset
TRAIN_N = 20000
X_train = X_train[:TRAIN_N]

# word index  text
word_index = imdb.get_word_index()
index_word = {idx+3: w for w, idx in word_index.items()}
index_word[0], index_word[1], index_word[2], index_word[3] = "<PAD>", "<START>", "<OOV>", "<UNUSED>"
def ids_to_text(ids, cap=200):
    return " ".join(index_word.get(int(i), "<UNK>") if int(i) < max_features else "<UNK>" for i in ids[:cap])

# ---------------- Make decoder inputs (shifted right with <START>) ----------------
START_ID = 1
def make_decoder_inputs(x):
    B, T = x.shape
    dec = np.empty_like(x)
    dec[:, 0] = START_ID
    dec[:, 1:] = x[:, :-1]
    return dec

dec_train = make_decoder_inputs(X_train)
dec_test  = make_decoder_inputs(X_test)

# ---------------- tf.data with PAD masking ----------------
BATCH = 256

def make_ds(src, dec, shuffle=False):
    ds = tf.data.Dataset.from_tensor_slices((src, dec))
    if shuffle: ds = ds.shuffle(10000)
    ds = ds.batch(BATCH).prefetch(tf.data.AUTOTUNE)
    return ds

# Use a proper train/val split from training data (dont peek at test)
split = int(0.9 * len(X_train))
train_ds_raw = make_ds(X_train[:split],  dec_train[:split], shuffle=True)
val_ds_raw   = make_ds(X_train[split:], dec_train[split:], shuffle=False)

# ---------------- Model (predicts embeddings; teacher forcing) ----------------
EMBED_DIM  = 96
ENC_UNITS  = 96
DEC_UNITS  = 128
LATENT_DIM = 32  # try 24 or 16 later

enc_inp = layers.Input(shape=(maxlen,), dtype="int32", name="enc_inp")
dec_inp = layers.Input(shape=(maxlen,), dtype="int32", name="dec_inp")

# IMPORTANT: turn OFF Keras masking to avoid BroadcastTo mask errors
emb_layer = layers.Embedding(max_features, EMBED_DIM, mask_zero=False, name="emb")

# Encoder
enc_emb = emb_layer(enc_inp)                                     # (B,T,E)
enc_vec = layers.Bidirectional(layers.LSTM(ENC_UNITS))(enc_emb)  # (B,2*ENC_UNITS)
z       = layers.Dense(LATENT_DIM, name="codings")(enc_vec)      # (B,Latent)

# Decoder inputs: previous-token embeddings + latent repeated at all steps
dec_emb = emb_layer(dec_inp)                                     # (B,T,E)
z_rep   = layers.RepeatVector(maxlen)(z)                         # (B,T,Latent)
dec_cat = layers.Concatenate(axis=-1)([dec_emb, z_rep])          # (B,T,E+Latent)

# (optional) small normalization helps stability
# dec_cat = layers.LayerNormalization()(dec_cat)

# Decoder RNN and projection to embedding space
dec_h   = layers.LSTM(DEC_UNITS, return_sequences=True, dropout=0.15)(dec_cat)
pred_emb = layers.TimeDistributed(layers.Dense(EMBED_DIM), name="pred_emb")(dec_h)  # (B,T,E)

# Targets & PAD masking for loss (mask PAD positions = id 0)
@tf.function
def lookup_embeddings(x_batch):
    return emb_layer(x_batch)  # (B,T,E)

def to_supervised(ds):
    def _map(x_src, x_dec):
        y_true = tf.stop_gradient(lookup_embeddings(x_src))       # reconstruct src embeddings
        w = tf.cast(tf.not_equal(x_src, 0), tf.float32)           # 1 where not PAD
        return (x_src, x_dec), y_true, w                          # sample_weight masks PAD in loss
    return ds.map(_map, num_parallel_calls=tf.data.AUTOTUNE).prefetch(tf.data.AUTOTUNE)

train_ds = to_supervised(train_ds_raw)
val_ds   = to_supervised(val_ds_raw)

# ---------------- Nearest-neighbor decode (forbid specials) ----------------
E = emb_layer.get_weights()[0]                                  # (V,E)
E_norm = E / (np.linalg.norm(E, axis=1, keepdims=True) + 1e-8)
forbidden = {0, 1, 2, 3}  # <PAD>, <START>, <OOV>, <UNUSED>

def emb_to_ids(emb_seq):                                        # (T,E) -> (T,)
    v = emb_seq / (np.linalg.norm(emb_seq, axis=1, keepdims=True) + 1e-8)
    sims = np.dot(v, E_norm.T)                                  # (T,V)
    sims[:, list(forbidden)] = -1e9
    return sims.argmax(axis=-1)
